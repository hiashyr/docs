Создание моделей базы данных

В этом разделе мы создадим модели для нашего магазина цветов. У нас есть три приложения, и для каждого мы создадим соответствующие модели.

Настройка пользовательской модели

Сначала настроим кастомную модель пользователя в приложении user_auth.

user_auth/models.py:

 1from django.db import models
 2from django.contrib.auth.models import AbstractUser
 3from django.core.validators import RegexValidator
 4
 5
 6class CustomUser(AbstractUser):
 7    """Расширенная модель пользователя"""
 8    name_validator = RegexValidator(
 9        regex=r'^[а-яА-ЯёЁ\s\-]+$',
10        message='Разрешены только кириллица, пробел и тире'
11    )
12    login_validator = RegexValidator(
13        regex=r'^[a-zA-Z0-9\-]+$',
14        message='Разрешены только латиница, цифры и тире'
15    )
16
17    first_name = models.CharField(
18        max_length=150,
19        validators=[name_validator],
20        verbose_name='Имя'
21    )
22    last_name = models.CharField(
23        max_length=150,
24        validators=[name_validator],
25        verbose_name='Фамилия'
26    )
27    patronymic = models.CharField(
28        max_length=150,
29        blank=True,
30        validators=[name_validator],
31        verbose_name='Отчество'
32    )
33    username = models.CharField(
34        max_length=150,
35        unique=True,
36        validators=[login_validator],
37        verbose_name='Логин'
38    )
39    email = models.EmailField(unique=True, verbose_name='Email')
40
41    class Meta:
42        verbose_name = 'Пользователь'
43        verbose_name_plural = 'Пользователи'
44
45    def get_full_name(self):
46        """Полное имя пользователя"""
47        full_name = f"{self.last_name} {self.first_name}"
48        if self.patronymic:
49            full_name += f" {self.patronymic}"
50        return full_name.strip()
51
52    def __str__(self):
53        return self.get_full_name() or self.username

Обновление настроек для кастомной модели пользователя

Теперь нужно указать Django использовать нашу кастомную модель пользователя. Откройте flower_shop/settings.py и добавьте:

# flower_shop/settings.py

AUTH_USER_MODEL = 'user_auth.CustomUser'

INSTALLED_APPS = [
    # ...
    'user_auth',
    # ...
]

 29class Product(models.Model):
 30    """Модель товара"""
 31    name = models.CharField(
 32        max_length=200,
 33        verbose_name='Название',
 34        validators=[RegexValidator(
 35            regex=r'^[а-яА-ЯёЁ0-9\s\-]+$',

 44    year = models.IntegerField(verbose_name='Год выпуска', validators=[MinValueValidator(1900)])
 45    model = models.CharField(max_length=100, verbose_name='Модель')
 46    stock_quantity = models.PositiveIntegerField(default=0, verbose_name='Количество на складе')
 47    is_available = models.BooleanField(default=True, verbose_name='В наличии')
 48    created_at = models.DateTimeField(auto_now_add=True, verbose_name='Дата добавления')
 49    updated_at = models.DateTimeField(auto_now=True, verbose_name='Дата обновления')

 51    def clean(self):
 52        """Валидация модели"""
 53        if self.price < 0:
 54            raise ValidationError({'price': 'Цена не может быть отрицательной'})
 55        if self.stock_quantity < 0:
 56            raise ValidationError({'stock_quantity': 'Количество на складе не может быть отрицательным'})

 61        self.is_available = self.stock_quantity > 0
 62        super().save(*args, **kwargs)

10class Order(models.Model):
11    """Модель заказа"""
12    STATUS_CHOICES = [
13        ('new', 'Новый'),
14        ('confirmed', 'Подтвержден'),
15        ('cancelled', 'Отменен'),
16    ]
17
18    user = models.ForeignKey(User, on_delete=models.CASCADE, verbose_name='Пользователь')
19    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='new', verbose_name='Статус')
20    created_at = models.DateTimeField(auto_now_add=True, verbose_name='Дата создания')
21    updated_at = models.DateTimeField(auto_now=True, verbose_name='Дата обновления')
22    cancellation_reason = models.TextField(blank=True, verbose_name='Причина отмены')
23
24    class Meta:
25        verbose_name = 'Заказ'
26        verbose_name_plural = 'Заказы'
27        ordering = ['-created_at']
28
29    def __str__(self):
30        return f"Заказ #{self.id} - {self.user.get_full_name()}"
31
32    @property
33    def total_quantity(self):
34        """Общее количество товаров в заказе"""
35        return sum(item.quantity for item in self.orderitem_set.all())
36
37    @property
38    def total_price(self):
39        """Общая стоимость заказа"""
40        return sum(item.get_total_price() for item in self.orderitem_set.all())
41

43class OrderItem(models.Model):
44    """Модель элемента заказа"""
45    order = models.ForeignKey(Order, on_delete=models.CASCADE, verbose_name='Заказ')
46    product = models.ForeignKey(Product, on_delete=models.CASCADE, verbose_name='Товар')
47    quantity = models.PositiveIntegerField(verbose_name='Количество')
48    price = models.DecimalField(max_digits=10, decimal_places=2, verbose_name='Цена за единицу')

62class Cart(models.Model):
63    """Модель корзины"""
64    user = models.ForeignKey(User, on_delete=models.CASCADE, verbose_name='Пользователь')
65    product = models.ForeignKey(Product, on_delete=models.CASCADE, verbose_name='Товар')
66    quantity = models.PositiveIntegerField(verbose_name='Количество')
67    created_at = models.DateTimeField(auto_now_add=True, verbose_name='Дата добавления')
68    updated_at = models.DateTimeField(auto_now=True, verbose_name='Дата обновления')
69
70    class Meta:
71        verbose_name = 'Корзина'
72        verbose_name_plural = 'Корзины'
73        unique_together = ['user', 'product']
74
75    def __str__(self):
76        return f"{self.user.username} - {self.product.name}"
77
78    def get_total_price(self):
79        """Общая стоимость элемента корзины"""
80        return self.quantity * self.product.price
81
82
83@receiver(post_save, sender=Order)
84def update_stock_on_order_confirmation(sender, instance, created, **kwargs):
85    """Уменьшает количество товаров на складе при подтверждении заказа"""
86    if not created and instance.status == 'confirmed':
87        # Проверяем, не был ли заказ уже подтвержден ранее
88        if Order.objects.filter(id=instance.id, status='confirmed').exists():
89            # Уменьшаем количество товаров на складе
90            for item in instance.orderitem_set.all():
91                item.product.stock_quantity -= item.quantity
92                if item.product.stock_quantity < 0:
93                    item.product.stock_quantity = 0
94                item.product.save()

Создание миграций

После создания всех моделей нужно создать и применить миграции:

# Создание миграций для всех приложений
python manage.py makemigrations

# Применение миграций
python manage.py migrate

# Если нужно создать миграции для конкретного приложения
python manage.py makemigrations main
python manage.py makemigrations orders
python manage.py makemigrations user_auth

Особенности моделей

CustomUser: - Наследуется от AbstractUser - Валидация имен кириллицей - Валидация логина латиницей - Уникальный email

Product: - Автоматическая валидация цены и количества - Автоматическое обновление статуса наличия - Валидация названия кириллицей

Order: - Сигнал для обновления склада при подтверждении заказа - Вычисляемые свойства для общей стоимости и количества

Cart: - Уникальная связь пользователь-товар - Автоматический расчет стоимости
