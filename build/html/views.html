Создание представлений (Views)

В этом разделе мы создадим представления для нашего магазина цветов. Представления обрабатывают запросы пользователей и возвращают ответы. У нас есть три приложения с разными типами представлений.

Представления для основного приложения (main)

Эти представления отвечают за отображение страниц магазина: главная, каталог, товары, контакты.

main/views.py:

 1from django.shortcuts import render, get_object_or_404
 2from django.db.models import Q

 6def home(request):
 7    """Главная страница"""
 8    slider_images = SliderImage.objects.filter(is_active=True).order_by('order')

12        'popular_products': popular_products,

17def catalog(request):
18    """Страница каталога"""
19    products = Product.objects.filter(is_available=True)
20    categories = Category.objects.all()
21
22    # Фильтрация по категории
23    category_id = request.GET.get('category')
24    if category_id:
25        products = products.filter(category_id=category_id)
26
27    # Сортировка
28    sort_by = request.GET.get('sort')
29    if sort_by == 'year':
30        products = products.order_by('-year')
31    elif sort_by == 'name':
32        products = products.order_by('name')
33    elif sort_by == 'price':
34        products = products.order_by('price')
35    else:
36        products = products.order_by('-created_at')  # По умолчанию по новизне
37
38    context = {
39        'products': products,
40        'categories': categories,
41        'current_category': category_id,
42        'current_sort': sort_by,
43    }
44    return render(request, 'main/catalog.html', context)
45

47def product_detail(request, product_id):

Особенности представлений main:

  • home: Отображает слайдер и популярные товары

  • catalog: Поддерживает фильтрацию по категориям и сортировку

  • product_detail: Показывает детальную информацию о товаре и похожие товары

  • contacts: Отображает контактную информацию

Представления для приложения заказов (orders)

Эти представления управляют корзиной и заказами.

orders/views.py:

 12@login_required
 13def cart_view(request):
 14    """Корзина пользователя"""
 15    cart_items = Cart.objects.filter(user=request.user)
 16    total_price = sum(item.get_total_price() for item in cart_items)
 17
 18    context = {
 19        'cart_items': cart_items,
 20        'total_price': total_price,
 21    }
 22    return render(request, 'orders/cart.html', context)

 24
 25@login_required
 26@require_http_methods(["POST"])
 27def add_to_cart(request, product_id):
 28    """Добавление товара в корзину"""
 29    product = get_object_or_404(Product, id=product_id, is_available=True)
 30    quantity = int(request.POST.get('quantity', 1))
 31
 32    if quantity > product.stock_quantity:
 33        return JsonResponse({
 34            'success': False,
 35            'message': f'Недостаточно товара на складе. Доступно: {product.stock_quantity}'
 36        })
 37
 38    cart_item, created = Cart.objects.get_or_create(
 39        user=request.user,
 40        product=product,
 41        defaults={'quantity': quantity}
 42    )
 43
 44    if not created:
 45        if cart_item.quantity + quantity > product.stock_quantity:
 46            return JsonResponse({
 47                'success': False,
 48                'message': f'Недостаточно товара на складе. Доступно: {product.stock_quantity}'
 49            })
 50        cart_item.quantity += quantity
 51        cart_item.save()
 52
 53    return JsonResponse({
 54        'success': True,
 55        'message': 'Товар добавлен в корзину'
 56    })
 57
 58
 59@login_required
 60@require_http_methods(["POST"])
 61def update_cart_item(request, cart_item_id):
 62    """Обновление количества товара в корзине"""
 63    cart_item = get_object_or_404(Cart, id=cart_item_id, user=request.user)
 64    quantity = int(request.POST.get('quantity', 1))
 65
 66    if quantity <= 0:
 67        cart_item.delete()
 68        return JsonResponse({'success': True, 'message': 'Товар удален из корзины'})
 69
 70    if quantity > cart_item.product.stock_quantity:
 71        return JsonResponse({
 72            'success': False,
 73            'message': f'Недостаточно товара на складе. Доступно: {cart_item.product.stock_quantity}'
 74        })
 75
 76    cart_item.quantity = quantity
 77    cart_item.save()
 78
 79    return JsonResponse({'success': True, 'message': 'Количество обновлено'})

 91@login_required
 92def checkout(request):
 93    """Оформление заказа"""
 94    cart_items = Cart.objects.filter(user=request.user)
 95
 96    if not cart_items.exists():
 97        messages.warning(request, 'Корзина пуста')
 98        return redirect('cart')
 99
100    if request.method == 'POST':
101        form = OrderConfirmationForm(request.user, request.POST)
102        if form.is_valid():
103            with transaction.atomic():
104                # Создаем заказ
105                order = Order.objects.create(user=request.user)
106
107                # Добавляем товары в заказ
108                for cart_item in cart_items:
109                    OrderItem.objects.create(
110                        order=order,
111                        product=cart_item.product,
112                        quantity=cart_item.quantity,
113                        price=cart_item.product.price
114                    )
115
116                    # НЕ уменьшаем количество на складе - это будет происходить только при подтверждении заказа
117
118                # Очищаем корзину
119                cart_items.delete()
120
121                messages.success(request, 'Заказ успешно оформлен!')
122                return JsonResponse({'success': True, 'message': 'Заказ успешно оформлен!', 'redirect': '/auth/profile/'})
123    else:
124        form = OrderConfirmationForm(request.user)
125
126    total_price = sum(item.get_total_price() for item in cart_items)
127
128    context = {
129        'form': form,
130        'cart_items': cart_items,
131        'total_price': total_price,
132    }
133    return render(request, 'orders/checkout.html', context)
134
135
136@login_required
137@require_http_methods(["POST"])
138def cancel_order(request, order_id):
139    """Отмена заказа пользователем"""
140    order = get_object_or_404(Order, id=order_id, user=request.user, status='new')
141
142    with transaction.atomic():
143        # Возвращаем товары на склад
144        for item in order.orderitem_set.all():
145            item.product.stock_quantity += item.quantity
146            item.product.save()
147
148        # Отменяем заказ
149        order.status = 'cancelled'
150        order.cancellation_reason = 'Отменен пользователем'
151        order.save()
152
153    messages.success(request, 'Заказ отменен')
154    return redirect('profile')

Особенности представлений orders:

  • cart_view: Показывает содержимое корзины

  • add_to_cart: AJAX-функция для добавления товаров

  • update_cart_item: AJAX-функция для обновления количества

  • remove_from_cart: AJAX-функция для удаления товаров

  • checkout: Оформление заказа с транзакцией

  • cancel_order: Отмена заказа с возвратом товаров на склад

Представления для аутентификации (user_auth)

Эти представления управляют регистрацией, авторизацией и профилем пользователя.

user_auth/views.py:

 1from django.shortcuts import render, redirect
 2from django.contrib.auth import login, logout
 3from django.contrib.auth.decorators import login_required
 4from django.contrib import messages
 5from django.http import JsonResponse
 6from django.views.decorators.csrf import csrf_exempt
 7from django.views.decorators.http import require_http_methods
 8from .forms import RegistrationForm, LoginForm
 9
10
11def register_view(request):
12    """Регистрация пользователя"""
13    if request.method == 'POST':
14        form = RegistrationForm(request.POST)
15        if form.is_valid():
16            user = form.save()
17            login(request, user)
18            messages.success(request, 'Регистрация прошла успешно!')
19            return redirect('home')
20        # Ошибки валидации будут показаны в форме
21    else:
22        form = RegistrationForm()
23
24    return render(request, 'user_auth/register.html', {'form': form})
25
26
27def login_view(request):
28    """Авторизация пользователя"""
29    if request.method == 'POST':
30        form = LoginForm(data=request.POST)
31        if form.is_valid():
32            user = form.get_user()
33            login(request, user)
34            messages.success(request, 'Вы успешно авторизованы!')
35            return redirect('home')
36        # Ошибки валидации будут показаны в форме
37    else:
38        form = LoginForm()
39
40    return render(request, 'user_auth/login.html', {'form': form})
41
42
43def logout_view(request):
44    """Выход из системы"""
45    logout(request)
46    messages.info(request, 'Вы вышли из системы')
47    return redirect('home')

Особенности представлений user_auth:

Декораторы и их назначение

Декораторы Django

Декоратор

Назначение

@login_required

Требует авторизации пользователя

@require_http_methods(["POST"])

Ограничивает доступ только для POST запросов

@csrf_exempt

Отключает CSRF защиту (используется осторожно!)

Типы представлений

Функциональные представления: - Простые в понимании - Прямолинейная логика - Идеальны для простых страниц

AJAX-представления: - Возвращают JSON ответы - Используются для динамических операций - Пример: добавление в корзину

Представления с формами: - Обрабатывают данные форм - Валидируют ввод пользователя - Пример: оформление заказа

Следующий шаг

После создания представлений переходите к <no title> чтобы создать HTML шаблоны для наших страниц.

Примечание

Совет: Используйте декораторы для защиты представлений и ограничения методов!

Предупреждение

Всегда используйте transaction.atomic() при работе с несколькими операциями с базой данных!
